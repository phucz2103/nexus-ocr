# Local OCR command reference

# Activate virtual environment

# Run router-service locally (GPU/default)
cd E:\Nexus project\ocr\apps\router_service
..\..\.venv\Scripts\python.exe -m uvicorn app.main:app --host 127.0.0.1 --port 8300

# Optional CPU fallback for VL if GPU memory is not enough
$env:ROUTER_SERVICE_VL_DEVICE="cpu"
..\..\.venv\Scripts\python.exe -m uvicorn app.main:app --host 127.0.0.1 --port 8300
